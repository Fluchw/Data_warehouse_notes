## 2. 装载脚本
若报错Error while compiling statement: FAILED: SemanticException Failed to get a spark session: org.apache.hadoop.hive.ql.metadata.HiveException: Failed to create Spark client for Spark session 
则在hadoop中/etc/hadoop/yarn-site.xml添加
```shell
 <!-- yarn容器允许分配的最大最小内存 -->
    <property>
        <name>yarn.scheduler.minimum-allocation-mb</name>
        <value>512</value>
    </property>
    <property>
        <name>yarn.scheduler.maximum-allocation-mb</name>
        <value>4096</value>
    </property>
    
    <!-- yarn容器允许管理的物理内存大小 -->
    <property>
        <name>yarn.nodemanager.resource.memory-mb</name>
        <value>4096</value>
    </property>
    
    <!-- 关闭yarn对物理内存和虚拟内存的限制检查 -->
    <property>
        <name>yarn.nodemanager.pmem-check-enabled</name>
        <value>false</value>
    </property>
    <property>
        <name>yarn.nodemanager.vmem-check-enabled</name>
        <value>false</value>
    </property>

<!-- hiveserver2的高可用参数，开启此参数可以提高hiveserver2的启动速度 -->

<property>
<name>hive.server2.active.passive.ha.enable</name>
<value>true</value>
</property>

<property>
  <name>hive.exec.dynamic.partition.mode</name>
  <value>nonstrict</value>
</property>
```
hdfs-site.xml
```shell
<property> 
<name>dfs.webhdfs.enabled</name> 
<value>true</value> 
</property>
```
分发
```shell
xsync yarn-site.xml
xsync hdfs-site.xml
```

然后重启集群
```shell
cd ~/bin
上传以下
```
